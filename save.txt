[BITS 16]
[ORG 0x7C00]

start:
    cli             ; Clear interrupts
    mov ax, 0x00
    mov ds, ax
    mov es, ax
    mov ss, ax
    mov sp, 0x7C00  ; Set up the stack pointer
    sti             ; Enable interrupts back

    mov si, msg     ; Load message pointer into SI

print:
    lodsb           ; Loads byte at ds:si to AL register and increments SI
    cmp al, 0
    je done         ; If null terminator, jump to done
    mov ah, 0x0E    ; BIOS teletype function
    int 0x10        ; Call BIOS interrupt
    jmp print

done:
    cli
    hlt             ; Stop further CPU execution

msg: db 'Hello World!', 0

times 510-($-$$) db 0   ; Pad remainder of boot sector with zeros
dw 0xAA55               ; Boot signature (little-endian for aa55)
